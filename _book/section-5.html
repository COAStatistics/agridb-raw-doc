5.4 檢查欄位最大長度

注意：下列函數必須讓資料為tibble格式才可運作

acd <- as_tibble(acd)
index <- 1
max_nchar <- function(x) {max(nchar(x))}
for (i in 1:dim(acd)[2]) {
  temp <- acd[,i]
  temp <- apply(temp, 2, max_nchar)
  if (index == 1) {
    table_nchar <- temp
    index <- 0
  } else table_nchar <- c(table_nchar, temp)
}
table_nchar
